    Quick start
  </h2>
  <p className="text-base text-gray-600 dark:text-gray-400 mb-8">
    Get up and running with AgentSpace in minutes
  </p>

  <Steps>
    <Step title="Install dependencies">
      Clone the repository and install the required packages:
      
      ```bash
      git clone https://github.com/nielchaudhary/agentspace.git
      cd agentspace
      npm install
      ```
    </Step>
    
    <Step title="Configure environment">
      Set up your OpenAI API key in a `.env` file:
      
      ```bash
      OPENAI_API_KEY=your_api_key_here
      ```
    </Step>
    
    <Step title="Start the server">
      Run the development server:
      
      ```bash
      npm run dev
      ```
      
      The server will start at `http://localhost:8090`
    </Step>
    
    <Step title="Make your first request">
      Test the translation agent:
      
      ```bash
      curl http://localhost:8090/agents/translate \
        -H "Content-Type: application/json" \
        -d '{"question": "Translate hello to Spanish"}'
      ```
      
      <Accordion title="Example response">
        ```json
        {
          "answer": "The Spanish translation of 'hello' is 'hola'."
        }
        ```
      </Accordion>
    </Step>
  </Steps>
</div>
